		{
			if((i%j)==0)
				break;
		}

		{
			if(j>(i/j)) 
				printf("%d 是质数\n",i);
			else printf("%d 不是质数\n",i);
		}
		printf("%d: %d:\n",i,j);

			break;
		}

	}

}
